.paraNav {

    transition: top 0.4s ease;
}

.paraNav.hide {
    top: -220px;
}

.paraNav li a {
    width: auto;
    padding: 5px 20px;
    width: 30px;
    height: 30px;
}

.paraNav li.active a {
    border-radius: 30px;
}

/* GSAP 한글자씩 나타나기 */
.content-item .desc span {
    opacity: 0;
    display: inline-block;
    min-width: 1vw;
    transform: translateY(100px) rotate(90deg) scale(2);
}

//글씨 쪼개기(여러개) each()
$(".content-item .desc").each(function () {
    let text = $(this).text();
    let split = text.split('').join("</span><span aria-hidden='true'>");
    split = "<span> aria-hidden='true'>" + split + "</span>";
    $(this).html(split).attr("aria-label", text);
});

    //each()
    $(".content-item .desc").each(function (index) {
        if ($(window).scrollTop() > $(this).offset().top - $(window).height() / 2) {
            $(this).find(".desc span").each(i => {
                setTimeout(() => {
                    $(this).find(".desc span").eq(i).addClass("show");
                }, 50 * i);
            });
        }
    });

    //gsap
    $(".content-item").each(function (index) {
        if ($(window).scrollTop() > $(this).offset().top) {
            gsap.to($(this).find(".desc span"), {
                duration: 0.5,
                opacity: 1,
                stagger: .05,
                y: 0,
                ease: "power3.out",
                rotation: 0,
                scale: 1
            });
        }
    });

    //버튼용
    $(".content-item").each(function(index, element){
        // console.log(index);
        // console.log(element);
        if($(window).scrollTop() >= $(this).offset().top - 2){
            $(".paraNav ul li").eq(index).addClass("active").siblings().removeClass("active");
        }
    });

    //출력용
    $(".paraScroll span").text(parseInt($(window).scrollTop()));
});
